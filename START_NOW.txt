━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Follow the guide: GOOGLE_CLOUD_SETUP.md

  Quick steps:
  1. Visit console.cloud.google.com
  2. Create project → Copy Project ID
  3. Enable 2 APIs (Text-to-Speech, YouTube Data)
  4. Create Service Account → Download JSON
  5. Create OAuth credentials → Download JSON
  6. Move both JSONs to config/credentials/
  7. Update .env with Project ID
  8. Set YOUTUBE_UPLOAD_ENABLED=true

  Then videos will auto-upload to YouTube Shorts!
